AI Connector (Python)

# ai_connector.py
from ayhem_core import ExperienceManager

def run_ai(experience_name):
    em = ExperienceManager()
    em.load(experience_name)
    em.process()
    return em.get_output()

if __name__ == "__main__":
    result = run_ai("Smart_Advanced_v2")
    print(result)
